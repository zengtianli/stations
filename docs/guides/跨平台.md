# 五、跨平台开发黄金法则 - 详细示例说明

## 5.1 五大核心原则

### 原则 1：路径灵活性

#### 1.1 不硬编码分隔符

**❌ 错误示例：**
```python
# 在 macOS 上写的代码
input_file = "data/input/user_data.csv"
output_file = "results/2024/report.txt"

# 在 Windows 上会出错
config_path = "/Users/john/project/config.json"  # 绝对路径硬编码
```

**✅ 正确示例：**
```python
from pathlib import Path

# 方案一：使用 pathlib（最推荐）
input_file = Path("data") / "input" / "user_data.csv"
output_file = Path("results") / "2024" / "report.txt"

# 方案二：使用 os.path.join
import os
input_file = os.path.join("data", "input", "user_data.csv")
output_file = os.path.join("results", "2024", "report.txt")

# 在 macOS 上结果：data/input/user_data.csv
# 在 Windows 上结果：data\input\user_data.csv
```

#### 1.2 不假设运行位置

**❌ 错误示例：**
```python
import os

# 假设用户必须在项目根目录运行脚本
current_dir = os.getcwd()  # 如果用户从别处运行，这里会是错误的路径
data_file = os.path.join(current_dir, "input.csv")

with open(data_file, "r") as f:
    data = f.read()
```

**✅ 正确示例：**
```python
import argparse
from pathlib import Path

# 方案一：通过命令行参数让用户指定
parser = argparse.ArgumentParser(description="数据处理工具")
parser.add_argument("data_dir", help="数据文件所在目录的完整路径")
parser.add_argument("--output", default="output", help="输出目录（可选）")
args = parser.parse_args()

data_path = Path(args.data_dir)
input_file = data_path / "input.csv"

with open(input_file, "r", encoding="utf-8") as f:
    data = f.read()

# 方案二：基于脚本自身位置定位文件
script_dir = Path(__file__).parent  # 脚本所在目录
data_dir = script_dir / "data"
input_file = data_dir / "input.csv"
```

**使用示例：**
```bash
# Windows 用户可以从任意位置运行
python C:\projects\tool\main.py D:\my_data

# macOS 用户也可以从任意位置运行
python3 ~/projects/tool/main.py ~/Documents/my_data
```

### 原则 2：包内相对引用

#### 2.1 使用 `.` 进行模块导入

**项目结构示例：**
```
my_project/
├── main.py
└── src/
    ├── __init__.py
    ├── utils/
    │   ├── __init__.py
    │   ├── file_handler.py
    │   └── logger.py
    └── models/
        ├── __init__.py
        ├── calculator.py
        └── validator.py
```

**❌ 错误示例（`src/models/calculator.py`）：**
```python
# 使用绝对导入 - 依赖项目根目录在 sys.path 中
from src.utils.logger import Logger
from src.models.validator import Validator

class Calculator:
    def __init__(self):
        self.logger = Logger()
        self.validator = Validator()
```

**问题**：
- 在 Windows 上，如果项目路径包含中文或特殊字符，可能导入失败
- 将项目作为子模块使用时会出错
- 不同机器的 Python 环境配置差异会导致找不到模块

**✅ 正确示例（`src/models/calculator.py`）：**
```python
# 使用相对导入
from ..utils.logger import Logger  # 向上一层，再进入 utils
from .validator import Validator   # 同级目录

class Calculator:
    def __init__(self):
        self.logger = Logger()
        self.validator = Validator()
```

#### 2.2 创建自包含单元

**`src/models/__init__.py` 完整示例：**

**❌ 错误示例：**
```python
# 依赖外部路径配置
from src.models.calculator import Calculator
```

### 原则 3：依赖明确化

#### 3.1 通过参数传递路径

**❌ 错误示例：**
```python
# 配置硬编码在代码里
INPUT_DIR = "/Users/john/Desktop/data"
OUTPUT_DIR = "/Users/john/Desktop/results"

def process_data():
    files = os.listdir(INPUT_DIR)
    # ... 处理逻辑
    with open(f"{OUTPUT_DIR}/report.txt", "w") as f:
        f.write("结果")
```

**✅ 正确示例：**
```python
import argparse
from pathlib import Path

def process_data(input_dir: Path, output_dir: Path):
    """
    处理数据的核心函数
    
    Args:
        input_dir: 输入文件目录
        output_dir: 输出文件目录
    """
    files = list(input_dir.glob("*.csv"))
    
    # 确保输出目录存在
    output_dir.mkdir(parents=True, exist_ok=True)
    
    report_file = output_dir / "report.txt"
    with open(report_file, "w", encoding="utf-8") as f:
        f.write(f"处理了 {len(files)} 个文件")

def main():
    parser = argparse.ArgumentParser(description="数据处理工具")
    parser.add_argument(
        "input_dir",
        type=Path,
        help="包含 CSV 文件的输入目录"
    )
    parser.add_argument(
        "-o", "--output",
        type=Path,
        default=Path("output"),
        help="输出目录（默认：./output）"
    )
    parser.add_argument(
        "--verbose",
        action="store_true",
        help="显示详细处理信息"
    )
    
    args = parser.parse_args()
    
    # 验证输入目录是否存在
    if not args.input_dir.exists():
        print(f"❌ 错误：输入目录不存在：{args.input_dir}")
        return 1
    
    process_data(args.input_dir, args.output)
    print("✅ 处理完成！")
    return 0

if __name__ == "__main__":
    exit(main())
```

**使用示例：**
```bash

python main.py "D:\项目数据" -o "D:\结果"

# macOS
python3 main.py ~/Documents/data -o ~/Documents/results
```

### 原则 4：友好错误处理

#### 4.1 预见 `IOError` 和 `ImportError`

**❌ 错误示例：**
```python
# 直接导入，失败就崩溃
from src.models import Calculator
import pandas as pd

# 直接读文件，找不到就崩溃
data = open("input.csv").read()
```

**✅ 正确示例：**
```python
import sys
from pathlib import Path

# 1. 处理导入错误
try:
    from src.models import Calculator
except ImportError as e:
    print("❌ 错误：无法导入 Calculator 模块")
    print(f"   详细信息：{e}")
    print("\n💡 可能的解决方案：")
    print("   1. 确认当前目录结构是否正确")
    print("   2. 检查是否缺少 __init__.py 文件")
    print("   3. 尝试从项目根目录运行脚本")
    sys.exit(1)

try:
    import pandas as pd
except ImportError:
    print("❌ 错误：未安装 pandas 库")
    print("\n💡 请运行以下命令安装：")
    print("   pip install pandas")
    sys.exit(1)

# 2. 处理文件读取错误
def load_data(file_path: Path):
    """安全地读取数据文件"""
    try:
        if not file_path.exists():
            raise FileNotFoundError(f"文件不存在：{file_path}")
        
        if not file_path.is_file():
            raise ValueError(f"路径不是文件：{file_path}")
        
        with open(file_path, "r", encoding="utf-8") as f:
            data = f.read()
        
        return data
    
    except FileNotFoundError as e:
        print(f"❌ 文件错误：{e}")
        print(f"💡 请检查文件路径是否正确：{file_path.absolute()}")
        return None
    
    except PermissionError:
        print(f"❌ 权限错误：无法读取文件 {file_path}")
        print("💡 请检查文件访问权限")
        return None
    
    except UnicodeDecodeError:
        print(f"❌ 编码错误：文件可能不是 UTF-8 编码")
        print("💡 尝试用记事本打开，另存为 UTF-8 编码")
        return None
    
    except Exception as e:
        print(f"❌ 未知错误：{type(e).__name__}: {e}")
        return None
```

#### ✅ 检查项 1：使用 pathlib 或 os.path.join()

**自查代码：**
```python
# 使用正则搜索可疑的硬编码路径
import re

def audit_hardcoded_paths(file_path):
    """审计代码中的硬编码路径"""
    with open(file_path, "r", encoding="utf-8") as f:
        content = f.read()
    
    # 查找可疑模式
    patterns = [
        r'"[A-Za-z]:\\[^"]+"',      # Windows 绝对路径 "C:\..."
        r'"/[^"]+"',                 # Unix 绝对路径 "/..."
        r'"[^"]+/[^"]+"',            # 包含 / 的路径
        r'"[^"]+\\[^"]+"',           # 包含 \ 的路径
    ]
    
    issues = []
    for pattern in patterns:
        matches = re.findall(pattern, content)
        issues.extend(matches)
    
    return issues

# 运行审计
suspicious_paths = audit_hardcoded_paths("main.py")
if suspicious_paths:
    print("⚠️  发现可疑的硬编码路径：")
    for path in suspicious_paths:
        print(f"   {path}")
```

### 5.2.2 模块导入

#### ✅ 检查项 1：包内使用相对导入

**审计脚本：**
```python
import os
import re

def audit_imports(directory):
    """审计项目中的导入语句"""
    issues = []
    
    for root, dirs, files in os.walk(directory):
        # 跳过虚拟环境和缓存
        dirs[:] = [d for d in dirs if d not in ['venv', '.venv', '__pycache__']]
        
        for file in files:
            if not file.endswith('.py'):
                continue
            
            file_path = os.path.join(root, file)
            with open(file_path, 'r', encoding='utf-8') as f:
                content = f.read()
            
            # 查找项目内的绝对导入
            pattern = r'from (src|app|project)\.'
            matches = re.findall(pattern, content)
            
            if matches:
                issues.append({
                    'file': file_path,
                    'matches': matches
                })
    
    return issues

# 运行审计
issues = audit_imports('src/')
if issues:
    print("⚠️  发现需要改为相对导入的地方：")
    for issue in issues:
        print(f"\n📄 {issue['file']}")
        print(f"   发现 {len(issue['matches'])} 处绝对导入")
```

```python
import ast
import os

class OpenCallVisitor(ast.NodeVisitor):
    """AST 访问者，查找所有 open() 调用"""
    def __init__(self):
        self.issues = []
        self.line_number = 0
    
    def visit_Call(self, node):
        # 检查是否是 open() 函数调用
        if isinstance(node.func, ast.Name) and node.func.id == 'open':
            # 检查是否有 encoding 参数
            has_encoding = any(
                kw.arg == 'encoding' for kw in node.keywords
            )
            
            if not has_encoding:
                self.issues.append(node.lineno)
        
        self.generic_visit(node)

def audit_file_operations(file_path):
    """审计文件中的 open() 调用"""
    with open(file_path, 'r', encoding='utf-8') as f:
        source = f.read()
    
    try:
        tree = ast.parse(source)
        visitor = OpenCallVisitor()
        visitor.visit(tree)
        return visitor.issues
    except SyntaxError:
        return []

# 扫描所有 Python 文件
for root, dirs, files in os.walk('src/'):
    for file in files:
        if file.endswith('.py'):
            file_path = os.path.join(root, file)
            issues = audit_file_operations(file_path)
            
            if issues:
                print(f"\n⚠️  {file_path}")
                print(f"   第 {', '.join(map(str, issues))} 行的 open() 缺少 encoding 参数")
```

```python
import re
import os

def audit_system_commands(file_path):
    """审计代码中的系统命令调用"""
    with open(file_path, 'r', encoding='utf-8') as f:
        content = f.read()
    
    # 危险模式
    dangerous_patterns = [
        (r'os\.system\s*\([^)]*\)', 'os.system()'),
        (r'subprocess\.(run|call|Popen)\s*\([^)]*["\'](?:ls|cp|mv|rm|grep|cat|pwd)["\']', 'shell命令'),
    ]
    
    issues = []
    for pattern, desc in dangerous_patterns:
        matches = re.finditer(pattern, content)
        for match in matches:
            line_num = content[:match.start()].count('\n') + 1
            issues.append({
                'line': line_num,
                'type': desc,
                'code': match.group()
            })
    
    return issues

# 扫描
for root, dirs, files in os.walk('src/'):
    for file in files:
        if file.endswith('.py'):
            file_path = os.path.join(root, file)
            issues = audit_system_commands(file_path)
            
            if issues:
                print(f"\n⚠️  {file_path}")
                for issue in issues:
                    print(f"   第 {issue['line']} 行：{issue['type']}")
                    print(f"   代码：{issue['code'][:60]}...")
```

#### ✅ 检查项 2：替换为 Python 跨平台模块

**替换对照表和示例：**

| 系统命令 | 危险代码 | 跨平台替代方案 |
|:---|:---|:---|
| **ls / dir** | `os.system('ls -l')` | `os.listdir()` 或 `Path.iterdir()` |
| **cp / copy** | `os.system('cp src dst')` | `shutil.copy()` |
| **mv / move** | `os.system('mv old new')` | `shutil.move()` |
| **rm / del** | `os.system('rm file')` | `os.remove()` 或 `Path.unlink()` |
| **mkdir** | `os.system('mkdir dir')` | `os.makedirs()` 或 `Path.mkdir()` |
| **cat / type** | `os.system('cat file')` | `open().read()` |
| **find** | `os.system('find .')` | `Path.rglob()` |

```python
    """跨平台的文件操作"""
    
    # 1. 列出文件
    current_dir = Path(".")
    files = list(current_dir.iterdir())
    
    # 写入文件列表
    with open("files.txt", "w", encoding="utf-8") as f:
        for file in files:
            size = file.stat().st_size if file.is_file() else 0
            f.write(f"{file.name}\t{size}\n")

    dest_dir.mkdir(exist_ok=True)  # 确保目录存在
```
