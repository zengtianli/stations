`~/Dev/devtools/lib/tools/repo_map_gen.py`： ```python #!/usr/bin/env python3 """扫本地 ~/Dev 生成 repo-map.json。 实际 layout： ~/Dev/stations/* 生产站群 ~/Dev/tools/* 工具基建 (configs, cc-configs, scripts) 分层目录：每个下面每个含 .git 的子目录是一个 repo LAYERED_CATEGORIES = ("stations", "tools", "labs", "content", "migrated") # 根下直接是 repo 的（非分层） ROOT_REPOS = ( "devtools", "doctools", "mactools", "clashx", "dotfiles", "raycast", "hammerspoon", "dockit-raycast", "scripts", "vps", "international-assets", "cc-configs", ) def git_remote(d: Path) -> str | None: try: r = subprocess.run( ["git", "-C", str(d), "config", "--get", "remote.origin.url"], capture_output=True, text=True, check=True, ) return r.stdout.strip() or None except subprocess.CalledProcessError: return None def scan() -> dict: out: dict = {} # 分层 for cat in LAYERED_CATEGORIES: base = DEV_ROOT / cat if not base.exists(): continue entries = {} for d in sorted(base.iterdir()): if not d.is_dir() or not (d / ".git").exists(): continue entries[d.name] = { "path": f"~/Dev/{cat}/{d.name}", "remote": git_remote(d), } if entries: out[cat] = entries # 根下 root_entries = {} for name in ROOT_REPOS: d = DEV_ROOT / name if not d.is_dir() or not (d / ".git").exists(): continue root_entries[name] = { "path": f"~/Dev/{name}", "remote": git_remote(d), } if root_entries: out["root"] = root_entries return out def main(): data = scan() n = file=sys.stderr) return OUT_PATH.write_text(text, encoding="utf-8") print(f"Wrote {OUT_PATH} ({n} repos across {len(data)} categories)") if __name__ == "__main__": main() ``` **新建** slash command
